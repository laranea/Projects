def model_retriver(factor):
    y_cols=[factor]
    X_cols=[]
    all_cols=[factor]

    for i in range(1,5): 
        col_name=str(factor)+"_"+str(i)
        X_cols.append(col_name)
        all_cols.append(col_name)

    df=us[all_cols]
    y_7=us[y_cols]
    X_7=us[X_cols]
    X_7=sm.add_constant(X_7)
    model_7=sm.OLS(y_7.astype(float), X_7.astype(float)).fit()
#     model_7=0
    return df, model_7, X_7
